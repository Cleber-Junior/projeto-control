create database controlf;

use controlf;

create table usuario (
    id_usuario int (6) auto_increment primary key,
    email varchar(30) not null,
    senha varchar(255) not null,
    nome varchar(50) not null,
    username varchar(25) not null,
    cpf char(14) not null,
    chave varchar(255),
    confirmacao int (1),
    img_usuario varchar(255) default 'foto.png' not null
);

create table admin(
    codAdmin varchar(3),
    nome varchar(50),
    senha varchar(255)
);

create table jogos(
	id_jogo int (6) auto_increment primary key,
    nome_jogo varchar(60) not null,
    sinopse_jogo varchar(3500) not null,
    preco_jogo Double(6,2),
    categoria varchar(50) not null,
    img_jogo varchar(50) default 'logo.png' not null,
    chave varchar(25) not null,
    quantidade int(255)
);

create table carrinho(
    id_carrinho int (6) auto_increment primary key,
    id_usuario integer not null,
    id_jogo integer not null,
    foreign key (id_usuario) references usuario(id_usuario),
    foreign key (id_jogo) references jogos(id_jogo),
    preco_total numeric(6,2)
);

create table compra(
    id_compra int (6) auto_increment primary key,
    data_compra Date,
    preco_compra Double
);

create table Biblioteca(
    nome_jogo varchar(60),
    chave_jogo varchar(25),
    img_jogo varchar(50) default 'logo.png' not null
);

create table listaDesejos(
    id_usuario integer not null,
    id_jogo integer not null,
    foreign key (id_usuario) references usuario(id_usuario),
    foreign key (id_jogo) references jogos(id_jogo),
    nome_jogo varchar (60)
);

create table categoria(
    id_categoria int (6) auto_increment primary key,
    tipo_cat varchar(50) not null
);
